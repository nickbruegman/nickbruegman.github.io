There is an age old argument in the world of UFC. Who truly is the best of the best? Many talk about the title defenses of champions, the knockout power of strikers, or the dominance of elite grapplers to debate the Greatest of All-Time. With the multiple weightclasses and various skill sets the sport displays, its difficult to truly compare fighters amoung weightclasses. However, I think it can be simplified. Just look at the stats.

When looking at http://ufcstats.com we can breakdown each fighters career and what they have accomplished. I think we can all agree, to be considered the GOAT of a sport, you have to have a championship under your belt. So letâ€™s look at all the present UFC champions as well as the past champions. (The past champions in this dataset all have atleast 1 title defense, current champions may not have a defense but that could be coming in their next fight)

library(dplyr)
library(tidyverse)
library(ggrepel)
library(ggplot2)
library(ggbeeswarm)

stats <- ufc.stats::ufc_stats

trial <- stats |> 
  separate(
    time, into=c("Minutes", "Seconds"), sep=":") |> 
  mutate(
    Last_Round_Time = (as.numeric(Minutes)*60) + as.numeric(Seconds))

clean1 <- trial |>
  group_by(fighter) |>
  summarize(
    Total_Fights = n_distinct(id),
    Rounds_Fought = n(),
    Sig_Strikes_Landed = sum(significant_strikes_landed),
    Sig_Strikes_Att = sum(significant_strikes_attempted),
    Tot_Strikes_Landed = sum(total_strikes_landed),
    Tot_Strikes_Att = sum(total_strikes_attempted),
    Takedowns = sum(takedown_successful),
    Takedown_Att = sum(takedown_attempted),
    Knockdowns = sum(knockdowns),
    Submission_Att = sum(submission_attempt),
    Reversals = sum(reversals),
    Head_Strikes = sum(head_landed),
    Head_Att = sum(head_attempted),
    Body_Strikes = sum(body_landed),
    Body_Att = sum(body_attempted),
    Leg_Strikes = sum(leg_landed),
    Leg_Att = sum(leg_attempted),
    Distance_Strikes = sum(distance_landed),
    Distance_Att = sum(distance_attempted),
    Clinch_Strikes = sum(clinch_landed),
    Clinch_Att = sum(clinch_attempted),
    Ground_Strikes = sum(ground_landed),
    Ground_Att = sum(ground_attempted),
    Time_Finished = sum(Last_Round_Time)
    ) |>
  mutate(
    Sig_Strike_PCT = (Sig_Strikes_Landed/Sig_Strikes_Att)*100,
    TOT_Strike_PCT = (Tot_Strikes_Landed/Tot_Strikes_Att)*100,
    Takedown_PCT = (Takedowns/Takedown_Att)*100,
    Head_Strikes_PCT = (Head_Strikes/Head_Att)*100,
    Body_Strikes_PCT = (Body_Strikes/Body_Att)*100,
    Leg_Strikes_PCT = (Leg_Strikes/Leg_Att)*100,
    AVG_Round_Ended = Rounds_Fought/Total_Fights,
    AVG_Sec_Finished = Time_Finished/Rounds_Fought)

clean2 <- trial |>
  filter(round == "round 1") |> 
  group_by(fighter) |> 
  summarize(
    Total_Fights = n_distinct(id),
    Wins = sum(winner == "W"),
    Losses = sum(winner == "L"),
    Draws = sum(winner == "D"),
    No_Contest = sum(winner == "NC"),
    CNC = sum(result == "Could Not Continue"),
    KO_TKO_W = sum(result == "KO/TKO" & winner == "W"),
    KO_TKO_L = sum(result == "KO/TKO" & winner == "L"),
    Submissions_W = sum(result == "Submission" & winner == "W"),
    Submissions_L = sum(result == "Submission" & winner == "L"),
    Unanimous_W = sum(result == "Decision - Unanimous" & winner == "W"),
    Unanimous_L = sum(result == "Decision - Unanimous" & winner == "L"),
    Split_W = sum(result == "Decision - Split" & winner == "W"),
    Split_L = sum(result == "Decision - Split" & winner == "L"),
    Majority_W = sum(result == "Decision - Majority"& winner == "W"),
    Majority_L = sum(result == "Decision - Majority"& winner == "L")
    ) |>
  mutate(
    Finishes = Submissions_W + KO_TKO_W,
    Finished = Submissions_L + KO_TKO_L,
    Win_PCT = (Wins / Total_Fights)*100,
    Finish_PCT = (Finishes / Wins)*100,
    Decision_W = Unanimous_W + Split_W + Majority_W
    )

ufc_careers <- merge(clean1, clean2, by = "fighter", all = TRUE)

champs <- c("Jon Jones", "Francis Ngannou", "Stipe Miocic", "Daniel Cormier", "Cain Velasquez", "Junior dos Santos", "Brock Lesnar", "Randy Couture", "Tim Sylvia", "Kevin Randleman", "Maurice Smith", "Alex Pereira", "Jan Blachowicz", "Lyoto Machida", "Quinton Jackson", "Chuck Liddell", "Tito Ortiz", "Israel Adesanya", "Anderson Silva", "Rich Franklin", "Chris Weidman", "Luke Rockhold", "Leon Edwards", "Kamaru Usman", "Tyron Woodley", "Robbie Lawler", "Georges St-Pierre", "Matt Hughes", "Pat Miletich", "Islam Makhachev", "Charles Oliveira", "Khabib Nurmagomedov", "Rafel Dos Anjos", "Anthony Pettis", "Benson Henderson", "Frankie Edgar", "BJ Penn", "Sean Sherk", "Jens Pulver", "Alexander Volkanovski", "Max Holloway", "Connor McGregor", "Jose Aldo", "Sean O'Malley", "Sean Strickland", "Aljamain Sterling", "Henry Cejudo", "TJ Dillashaw", "Dominick Cruz", "Renan Barao", "Brandon Moreno", "Deiveson Figueiredo", "Demetrious Johnson", "Alex Pereira", "Alexandre Pantoja", "Tom Aspinall")

chart_0 <- ufc_careers |>

    cols = ends_with("W"),

ggplot() +

Code 
goats <- ufc_careers |>
  group_by(fighter) |>
  filter(fighter %in% champs) |>
  filter(Win_PCT>50) |>
  filter(Finish_PCT>50) |>
  filter(Wins > 10)

chart_2a <- goats |>
  summarize(
    Sig_Strike_PCT,
    Takedown_PCT,
    Finishes) |>
  arrange(desc(Sig_Strike_PCT))

chart_2b <- goats |>
  summarize(
    Sig_Strike_PCT,
    Takedown_PCT,
    Finishes) |> 
  arrange(desc(Takedown_PCT))

t5_chart2a <- chart_2a |>
  top_n(5, wt= Sig_Strike_PCT)

t5_chart2b <- chart_2b |>
  top_n(5, wt= Takedown_PCT)

ggplot() + 
  geom_point(
    data= chart_2a,
    aes(x= Sig_Strike_PCT, y= Takedown_PCT, size= Finishes),
    alpha = .2) +
  geom_point(
    data= t5_chart2a,
    aes(x= Sig_Strike_PCT, y= Takedown_PCT, size= Finishes),
    color="red") +
  geom_point(
    data= t5_chart2b,
    aes(x= Sig_Strike_PCT, y= Takedown_PCT, size= Finishes),
    color="blue") +
  geom_vline(xintercept = 50) +
  geom_hline(yintercept = 50) +
  geom_text_repel(
    data=t5_chart2a,
